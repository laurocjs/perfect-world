var flkty = new Flickity('.carousel', { initialIndex: 2 }); var appendButton = document.querySelector('.button--append'); appendButton.addEventListener('click', function () { var cellElems = [makeCell(), makeCell()]; flkty.append(cellElems); }); var cellCount = flkty.cells.length; function makeCell() { cellCount++; var cell = document.createElement('div'); cell.className = 'carousel-cell'; cell.textContent = cellCount; return cell; }
